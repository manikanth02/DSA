```

idea - 1 :: Traverse Boundary of matrix and if any element is '0' then store in queue
and do Multi source in queue.Also maintain a visited matrix(all 0's) when we visits
and element mark 1 in visisted matrix


class Solution{
public:
    vector<vector<char>> fill(int row, int col, vector<vector<char>>& matrix){
        
      vector<vector<int>> visited(row,vector<int>(col,0));
      
      queue<pair<int,int>> q;
      
     vector<int> X = { -1, 0, +1,0};
     vector<int> Y = { 0, +1, 0, -1};
    
    //first Row
    for(int j = 0; j < col; j++){
        if(matrix[0][j] == 'O'){
            visited[0][j] = 1;
            q.push({0,j});
        }
    }
    
    //last Row
    if(row > 1){
    for(int j = 0; j < col; j++){
        if(matrix[row - 1][j] == 'O'){
            visited[row - 1][j] = 1;
            q.push({row - 1,j});
        }
    }
    }
    
    for(int i = 1; i < row - 1; i++){
        if(matrix[i][0] == 'O'){
            visited[i][0] = 1;
            q.push({i,0});
        }
    }
    
    if(col > 1){
        for(int i = 1; i < row - 1; i++){
        if(matrix[i][col - 1] == 'O'){
            visited[i][col - 1] = 1;
            q.push({i,col - 1});
        }
    }
    }
      
      while(!q.empty()){
          
          auto current = q.front();
          int x = current.first;
          int y = current.second;
          q.pop();
          
          for(int k = 0; k < 4; k++){
              int newX = x + X[k];
              int newY = y + Y[k];
              
              if(newX >= 0 && newX < row && newY >= 0 && newY < col){
                  if(visited[newX][newY] == 0 && matrix[newX][newY] == 'O'){
                      visited[newX][newY] = 1;
                      q.push({newX,newY});
                    }
                }
            }
       }
       
       for(int i = 0 ; i < row; i++){
           for(int j = 0 ; j < col; j++){
               if(matrix[i][j] == 'O' && visited[i][j] == 0){
                   matrix[i][j] = 'X';
               }
           }
       }
       
       return matrix;
        
        
    }
};
```
