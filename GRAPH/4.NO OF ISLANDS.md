idea - 1:: BFS Algorithms


class Solution {
public:
    vector<int> X = {-1,0,+1,0};
    vector<int> Y = {0,+1,0,-1};
    void bfs(int i, int j , vector<vector<char>> & matrix){
        queue<pair<int,int>>q;
        q.push({i,j});
        matrix[i][j] = '0';
        int row = matrix.size();
        int col = matrix[0].size();
        while(!q.empty()){
            auto current = q.front();
            q.pop();

            for(int k = 0; k < 4; k++){
                int newX = current.first + X[k];
                int newY = current.second + Y[k];

                if(newX < 0 || newX >= row || newY < 0 || newY >= col || matrix[newX][newY] == '0'){
                    continue;
                }
                q.push({newX,newY});
                matrix[newX][newY] = '0';

            }
        }

    }
    int numIslands(vector<vector<char>>& grid) {
        vector<vector<char>> matrix = grid;
        int row = matrix.size();
        int col = matrix[0].size();

        int count = 0;
        for(int i = 0; i < row; i++){
            for(int j = 0; j < col; j++){
                if(matrix[i][j] == '1'){
                    bfs(i,j,matrix);
                    count++;
                }
            }
        }
        return count;
    }
};

class Solution {
public:
    vector<int> X = {-1,0,+1,0};
    vector<int> Y = {0,+1,0,-1};
    void dfs(int i, int j , vector<vector<char>> & matrix){
        if(i < 0 || i >= matrix.size() || j < 0 || j >= matrix[0].size() || matrix[i][j] == '0'){
        return;
        }

        matrix[i][j] = '0';

        for(int k = 0; k < 4; k++){
            int newX = i + X[k];
            int newY = j + Y[k];
            dfs(newX,newY,matrix);
        }
    }
    int numIslands(vector<vector<char>>& grid) {
        vector<vector<char>> matrix = grid;
        int row = matrix.size();
        int col = matrix[0].size();

        int count = 0;
        for(int i = 0; i < row; i++){
            for(int j = 0; j < col; j++){
                if(matrix[i][j] == '1'){
                    dfs(i,j,matrix);
                    count++;
                }
            }
        }
        return count;
    }
};
