class Solution {
  public:
    string findOrder(string dict[], int n, int k) {
        // code here
        
        //Step 1:: genertae graph
        
        vector<int>adj[k];
        for(int i = 0; i < n - 1;i++){
            for(int j = 0; j < min(dict[i].size(),dict[i + 1].size()); j++){
            
                if(dict[i][j] == dict[i + 1][j]){
                    continue;
                }else{
                    adj[dict[i][j] - 'a'].push_back(dict[i + 1][j] - 'a');
                    break;
                }
            }
        }
        
        vector<int>indegree(k,0);
        for(int i = 0; i < k; i++){
            for(auto &ele : adj[i]){
                indegree[ele]++;
            }
        }
        string ans = "";
        queue<int>q;
        for(int i = 0; i < k; i++){
            if(indegree[i] == 0){
                q.push(i);
            }
        }
        
        while(!q.empty()){
            int curr = q.front();
            q.pop();
            ans += (char)(curr) + 'a';
            
            for(auto &ele :adj[curr]){
                indegree[ele]--;
                if(indegree[ele] == 0){
                    q.push(ele);
                }
            }
        }
        return ans;
    }
};
