PROBLEM:: Find the number of land cells in matrix for which 
we can not move out of Boundary.

SOLUTION::start traversing from Boundary 1's and make 0 to all 1's which is 
connected to 1.


class Solution {
  public:
    int numberOfEnclaves(vector<vector<int>> &matrix) {
        
    int row = matrix.size();
    int col = matrix[0].size();
      
    queue<pair<int,int>> q;
      
    vector<int> X = { -1, 0, +1,0};
    vector<int> Y = { 0, +1, 0, -1};
    
    //first Row
    for(int j = 0; j < col; j++){
        if(matrix[0][j] == 1){
            q.push({0,j});
        }
    }
    
    //last Row
    if(row > 1){
    for(int j = 0; j < col; j++){
        if(matrix[row - 1][j] == 1){
            q.push({row - 1,j});
        }
    }
    }
    
    for(int i = 1; i < row - 1; i++){
        if(matrix[i][0] == 1){
            q.push({i,0});
        }
    }
    
    if(col > 1){
        for(int i = 1; i < row - 1; i++){
        if(matrix[i][col - 1] == 1){
            q.push({i,col - 1});
        }
    }
    }
    
    
    while(!q.empty()){
        auto current = q.front();
        int x = current.first;
        int y = current.second;
        q.pop();
        matrix[x][y] = 0;
        
        for(int k = 0 ; k < 4; k++){
            int newX = x + X[k];
            int newY = y + Y[k];
            
            if(newX >= 0 && newX < row && newY >= 0 && newY < col &&
            matrix[newX][newY] == 1){
                q.push({newX,newY});
                matrix[newX][newY] = 0;;
            }
        }
    }
    
    int noOfEnclaves = 0;
    
    for(int i = 0; i < row; i++){
        for(int j = 0; j < col ; j++){
            if(matrix[i][j] == 1){
                noOfEnclaves++;
            }
        }
    }
    
    return noOfEnclaves;
    }
};
