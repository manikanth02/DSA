class Solution
{
  public:
    vector<int> findOrder(int N, int P, vector<vector<int>> prerequisites) 
    {
        //code here
         vector<int>indegree(N,0);
	    vector<int> adj[N];
	    
	    // Generate Graph
	    for(int i = 0; i < P; i++){
	        adj[prerequisites[i][1]].push_back(prerequisites[i][0]);
	    }   
	    
	    
	    for(int i = 0; i < N; i++){
	        for(auto &ele : adj[i]){
	            indegree[ele]++;
	        }
	    }
	    
	    queue<int>q;
	    for(int i = 0; i < N; i++){
	        if(indegree[i] == 0){
	            q.push(i);
	        }
	    }
	    
	    vector<int>topoSort;
	    
	    while(!q.empty()){
	        int curr = q.front();
	        q.pop();
	        topoSort.push_back(curr);
	        
	        for(auto &ele : adj[curr]){
	            indegree[ele]--;
	            if(indegree[ele] == 0){
	                q.push(ele);
	            }
	        }
	    }
	    
	    if(topoSort.size() == N){
	        return topoSort;
	    }
	    return {};
	
    }
};
