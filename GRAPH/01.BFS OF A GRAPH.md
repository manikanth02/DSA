## BFS OF GRAPH




```
vector<int>bfsOFGraph(vector<vector<int>> adj , int V){
    queue<int>q;
    vector<bool>visited(V + 1, false);
    q.push(1);
    visited[1] = true;
    vector<int>bfs;

    while(!q.empty()){
        int frontElement = q.front();
        q.pop();
        bfs.push_back(frontElement);

        for(auto &ele : adj[frontElement]){
            if(visited[ele] == false){
                q.push(ele);
                visited[ele] = true;
            }
        }
    }
    return bfs;
}
```
