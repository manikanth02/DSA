```
class Solution {
  public:
    int findPair(int n, int x, vector<int> &A) {
        // code here
        
        unordered_map<int,int>mp;
        
        for(auto num:A){
            mp[num]++;
            
            if( x == 0 && mp[num] > 1){
                return 1;
            }
        }
        
        //edge case
        if( x == 0)return -1;
        
        
        for(int i = 0; i < n; i++){
            if(mp.find(abs(x + A[i])) != mp.end()){
                return 1;
            }
        }
        return -1;
    }
};

```
