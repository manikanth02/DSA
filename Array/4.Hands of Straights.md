class Solution {
public:
    bool isNStraightHand(vector<int>& A, int groupSize) {

        map<int,int>freq;

        for(int num:A) freq[num]++;

        for (auto it = freq.begin(); it != freq.end(); ++it) {
            if (it->second > 0) {
                int count = it->second;
                for (int i = 0; i < groupSize; i++) {
                    if (freq[it->first + i] < count) {
                        return false;
                    }
                    freq[it->first + i] -= count;
                }
            }
        }

        return true;
    }
};

// 2->2
// 3->2
// 4->1
