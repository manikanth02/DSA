T.C::O(N^2)
S.C::O(1)

class Solution {
public:
    bool check(int x,vector<int>& A){
        int count = 0;
        int n = A.size();
        for(int i = 0; i < n; i++){
            if(A[i] >= x){
                count++;
            }
        }
        return (count == x);
    }
    int specialArray(vector<int>& A) {
      int n = A.size();
      for(int i = 0;i <= n; i++){
        if(check(i,A)){
            return i;
        }
      }
      return -1;  
    }
};

// PROBLEMS::Given an array,we have to check is there any numbers k such that there are exaclty k numbers whose value >= k;

class Solution {
public:
    int specialArray(vector<int>& A) {
        int n = A.size();
        vector<int>count(1001,0);
        for(int i = 0;i < n; i++){
            count[A[i]]++;
        }
        int total = n;
        for(int i = 0; i <= 1000; i++){
            if(i == total){
                return i;
            }
            total -= count[i];
        }
        return -1;
        
    }
};
